function resolveCollisions() { const boxes = Array.from(document.querySelectorAll('.word-box')); const iterations = 10; // Repeat a few times to settle cascading chain-reaction bumps for (let i = 0; i < iterations; i++) { let collisionsResolved = 0; for (let j = 0; j < boxes.length; j++) { for (let k = j + 1; k < boxes.length; k++) { const b1 = boxes[j]; const b2 = boxes[k]; const r1 = b1.getBoundingClientRect(); const r2 = b2.getBoundingClientRect(); // Check if they overlap const overlapX = Math.min(r1.right, r2.right) - Math.max(r1.left, r2.left); const overlapY = Math.min(r1.bottom, r2.bottom) - Math.max(r1.top, r2.top); if (overlapX > 0 && overlapY > 0) { collisionsResolved++; // Push along the axis with the shallowest overlap to keep movements natural if (overlapX < overlapY) { const pushX = overlapX / 2; const sign = r1.left + r1.width / 2 < r2.left + r2.width / 2 ? -1 : 1; adjustElementX(b1, sign * pushX); adjustElementX(b2, -sign * pushX); } else { const pushY = overlapY / 2; const sign = r1.top + r1.height / 2 < r2.top + r2.height / 2 ? -1 : 1; adjustElementY(b1, sign * pushY); adjustElementY(b2, -sign * pushY); } // Keep boxes within canvas boundaries during the push constrainToCanvas(b1); constrainToCanvas(b2); } } } // If a pass completes with zero overlaps, we can stop early if (collisionsResolved === 0) break; } }
